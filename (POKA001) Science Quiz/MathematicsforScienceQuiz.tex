% Problem environments
% Past Problems keep explicit historical labels. Expected Problems use an
% automatic counter that is reset independently in each problem category.
\newcommand{\problemheading}[1]{%
    \par\vspace{6pt}\noindent
    \textbf{Problem #1.}\hspace{0.5em}\ignorespaces
}

\newenvironment{problem}[1]
    {\problemheading{#1}}
    {\par\vspace{6pt}}

\newenvironment{pastproblem}[1]
    {\problemheading{#1}}
    {\par\vspace{6pt}}

\newcounter{expectedproblemsection}
\newcounter{expectedproblem}
\renewcommand{\theHexpectedproblem}{%
    \arabic{expectedproblemsection}.\arabic{expectedproblem}%
}
\newenvironment{expectedproblem}[1][]
    {%
        \refstepcounter{expectedproblem}%
        \if\relax\detokenize{#1}\relax
            \problemheading{\theexpectedproblem}%
        \else
            \problemheading{\theexpectedproblem.\ (#1)}%
        \fi
    }
    {\par\vspace{6pt}}

\newcommand{\expectedproblemsection}[1]{%
    \section*{#1}%
    \stepcounter{expectedproblemsection}%
    \setcounter{expectedproblem}{0}%
}

\newenvironment{solution}
    {\par\vspace{6pt}\noindent\textit{Solution.}\hspace{0.5em}\ignorespaces}
    {\par\nobreak\noindent\hfill$\square$\par\vspace{6pt}}

\newenvironment{hint}
    {\par\vspace{4pt}\noindent\textit{Hint.}\hspace{0.5em}\ignorespaces}
    {\par\vspace{4pt}}
